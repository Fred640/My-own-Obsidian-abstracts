##### Добавление пользователей:

после добавления первого sudo пользователя остальных пользователей можно добавлять и распределять по группам через созданного sudo пользователя в админ панели
###### sudo пользователь:
1) `python manage.py createsuperuser`
2) ввести имя
3) ввести почту
4) ввести пароль и повторить
##### Использование админ панели
регистрирование моделей 
```
@admin.register(<ModelName>)
class <ModelName>Admin(admin.ModelAdmin):
	list_display = ('id', "name", 'isPublished')     #поля которые буду отображаться на странице
	list_display_links = ["name"]                    #поля которые будут ссылками на страницу записи
	ordering = ['id']                                #поле по которому будут сортироваться записи на странице модели
	list_editable = []                               #список полей которые можно редоктировать 
	list_per_page = 4                                #количетво отображаемых записей на одной странице
	actions=[]                                       #действия с записями
	search_fields = ['name']                         #список полей по которым можно осуществлять поиск записей
	list_filter = ["cat__name"]                      #список полей по которым можно фильтровать записи
	fields = []                                      #поля которые будут доступны на форме при добавлении новой записи или редакировании старой записи
```

Добавление своих действий
```
@admin.register(People)
class PeopleAdmin(admin.ModelAdmin):
	list_display = ('id', "name", 'isPublished', 'brief_info')
	list_display_links = ["name"]
	ordering = ['id']
	actions = ['set_published', 'set_no_published']
	
	@admin.action(description="Опубликовать")
	def set_published(self, request, queryset):
		count = queryset.update(isPublished=1)
		self.message_user(request, f"изменено {count} записей")
		
	@admin.action(description="Скрыть")
	def set_no_published(self, request, quryset):
		count = quryset.update(isPublished=0)
		self.message_user(request, f"изменено {count} записей")
```
